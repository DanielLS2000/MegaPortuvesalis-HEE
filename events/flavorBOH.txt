#The Heretic King
country_event = {
	id = flavor_boh.2
	title = flavor_boh.2.t
	desc = flavor_boh.2.d
	picture = {	
		trigger = {
			has_dlc = "Emperor"
		}
		picture = HUSSITE_eventPicture
	}
	picture = {	
		trigger = {
			NOT = { has_dlc = "Emperor" }
		}
		picture = COURT_eventPicture
	}

	is_triggered_only = yes

	trigger = {
		OR = {
			tag = BOH
			was_tag = BOH
		}
		OR = {
			AND = {
				religion = catholic
				ruler_religion = hussite
			}
			religion = hussite
		}
		has_regency = no
		NOT = { has_ruler_flag = hussite_king }
		NOT = { is_religion_enabled = protestant }
		NOT = {
			check_variable = {
				which = hussite_changes_to_repent
				value = 3
			}
		}
		had_country_flag = {
			flag = first_hussite_king
			days = 182
		}
	}
	
	immediate = {
		hidden_effect = {
			set_ruler_flag = hussite_king
			if = {
				limit = {
					NOT = {
						check_variable = {
							which = hussite_changes_to_repent
							value = 1
						}
					}
				}
				set_variable = {
					which = hussite_changes_to_repent
					value = 1
				}					
			}
			else_if = {
				limit = {
					check_variable = {
						which = hussite_changes_to_repent
						value = 1
					}
				}
				change_variable = {
					which = hussite_changes_to_repent
					value = 1
				}
			}
		}
	}
	
	option = { # reconciliation
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0.1
				is_year = 1470
			}
		}
		name = flavor_boh.2.a
		if = {
			limit = { exists = PAP }
			add_papal_influence = 5
			PAP = {
				add_opinion = {
					who = root
					modifier = boh_papal_reconciled
				}
			}
		}
		add_reform_desire = -0.01
		if = {
			limit = {
				religion = hussite
			}
			change_religion = catholic
			every_owned_province = {
				limit = {
					religion = hussite
				}
				change_religion = catholic
			}
		}
		if = {
			limit = {
				ruler_religion = hussite
			}
			set_ruler_religion = catholic
		}
		if = {
			limit = {
				heir_religion = hussite
			}
			set_heir_religion = catholic
		}
	}
	option = {
		name = flavor_boh.2.b
		ai_chance = {
			factor = 10
		}
		add_ruler_modifier = {
			name = "boh_hussite_king"
			duration = -1
		}
		add_reform_desire = 0.005
		if = {
			limit = {
				religion = catholic
			}
			excommunicate = ROOT
		}
		else_if = {
			limit = {
				religion = hussite
			}
			custom_tooltip = rejected_reconciliation_tt
			set_ruler_flag = hussite_heretic # catholics can cleanse heresy against you for 15 years
		}
		if = {
			limit = {
				exists = PAP
			}
			PAP = { country_event = { id = flavor_boh.9 } }
		}
	}
}
